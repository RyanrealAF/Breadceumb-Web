# ARTIFACT 001: THE IMMUTABILITY MANDATE
# Cryptographic Hashing Script (SHA-256)
# This module provides a function to generate a cryptographic hash of the Evidence Grid data.
# This hash acts as the "lock" for the Dead-Man's Switch and proves the data integrity of
# the public record. A change of even one character in the input data will produce an entirely
# different hash, ensuring the public record is provably immutable.

import hashlib
import json
import time

def generate_breadcrumb_hash(data_structure):
    """
    Generates a unique SHA-256 hash for the input data structure.
    """
    if isinstance(data_structure, dict):
        normalized_data = json.dumps(data_structure, sort_keys=True, indent=None)
    elif isinstance(data_structure, str):
        normalized_data = data_structure
    else:
        normalized_data = str(data_structure)

    data_bytes = normalized_data.encode('utf-8')
    hasher = hashlib.sha256()
    hasher.update(data_bytes)
    return hasher.hexdigest()

    "declaration": "THE EVIDENCE GRID IS SECURE. NO EDITS AFTER LOCK.",
    "core_statement": "The truth is arithmetic. The reaction is the fuel.",
}

# Generate the initial lock hash
initial_hash = generate_breadcrumb_hash(EVIDENCE_GRID_MOCK)
print(f"--- LOCK INITIATED ({time.strftime('%Y-%m-%d %H:%M:%S UTC', time.gmtime(EVIDENCE_GRID_MOCK['timestamp_utc']))}) ---")
